## Query example

```javascript JavaScript
const axios = require("axios");

(async () => {
  const response = await axios.post(
    "https://api.fundstream.com/v1/purchases",
    {
      userId: "12345",
      amount: 100.00,
      currency: "USD",
      mobileNumber: "+1234567890",
      provider: "M-Pesa"
    },
    {
      headers: {
        "X-API-Key": "your_api_key"
      }
    }
  );

  console.log(response.data);
})();
```
